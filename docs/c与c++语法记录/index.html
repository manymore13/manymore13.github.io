指针的用法

#include <stdio.h>

int main(){
    // 一维数组
    int arr[] = {100,200,300};
    // 打印一维数组数据
    for(int i=0;i<3;i++){
        printf("数组 %d\n",arr[i]);
    }
    // 一维数组中第二个元素地址赋给int型p指针
    // 数组坐标从零开始
    int *p = &arr[1];
    // p指针往后移一个单位位置，一个单位是int4个字节
    // 这里打印的是数组的第三个元素
    printf("p=%d\n",*(++p));
    return 0;
}

数组 200
数组 300
p=300
